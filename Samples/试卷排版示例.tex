\item 若 $(x+1)^2=x^2+2x+1$，则一次项系数为（　）。
\begin{itemize}
\item[A.] $1$
\item[B.] $2$
\item[C.] $-1$
\item[D.] $-2$
\end{itemize}
\end{enumerate}
